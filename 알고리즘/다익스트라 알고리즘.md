## Solution

```python
import heapq

V, E = list(map(int, input().split()))
start = int(input())
graph = [[] for i in range(V + 1)]
for i in range(E):
    u, v, w = list(map(int, input().split()))
    graph[u].append([w, v])
dist = [float('inf')] * (V + 1)
visit = [0] * (V + 1)
dist[start] = 0
h = [[0, start]]
while h:
    price, idx = heapq.heappop(h)
    visit[idx] = 1
    for i in graph[idx]:
        w, v = i
        if dist[idx] + w < dist[v]:
            dist[v] = dist[idx] + w
    for i in graph[idx]:
        w, v = i
        if visit[v] == 0:
            heapq.heappush(h, [w, v])

for i in range(1, V + 1):
    if dist[i] == float('inf'):
        print("INF")
    else:
        print(dist[i])

```
