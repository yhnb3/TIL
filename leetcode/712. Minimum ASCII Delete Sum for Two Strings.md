## [712. Minimum ASCII Delete Sum for Two Strings](https://leetcode.com/problems/minimum-ascii-delete-sum-for-two-strings/)

```python
class Solution:
    def minimumDeleteSum(self, s1: str, s2: str) -> int:
        asciiSum = 0
        for s in s1:
            asciiSum += ord(s)
        for s in s2:
            asciiSum += ord(s)

        dp = [[0] * len(s1) for _ in range(len(s2))]

        for i in range(len(s2)):
            for j in range(len(s1)):
                if i == 0:
                    if s2[i] == s1[j]:
                        if j == 0:
                            dp[i][j] = ord(s2[i])
                        else:
                            dp[i][j] = max(dp[i][j-1], ord(s2[i]))
                    else:
                        dp[i][j] = dp[i][j-1]
                else:
                    if s2[i] == s1[j]:
                        if j == 0:
                            dp[i][j] = ord(s2[i])
                        else:
                            dp[i][j] = max(dp[i][j-1], dp[i-1][j-1] + ord(s2[i]), dp[i-1][j])
                    else:
                        dp[i][j] = max(dp[i][j-1], dp[i-1][j])
        return asciiSum - (dp[len(s2)-1][len(s1)-1] * 2)
```
